const puppeteer = require('puppeteer');

async function start() {
    const browser = await puppeteer.launch();
    const page = await browser.newPage();
    await page.goto("https://rahul-gupta.netlify.app");
    await page.screenshot({ path: "rahulPortfolio.png" , fullPage : true });
    await browser.close();
}

start();
